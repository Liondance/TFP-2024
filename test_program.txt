[
        <Z -> Z> chs := <Z> x -> minus(0)(x);
        <Z -> (Z -> Z)> plus := (<Z> x -> (<Z> y -> minus(x)(chs(y))));
        <Z> x := 7;
        <Z> y := 8;
        <Z> q := plus(x)(y);
        .show(q);                        -- imprime: q ==> 15
        <lazy Z> z := 'plus(x)(y)';
        .show(z);                        -- imprime: z ==> 15
        x := plus(x)(1);
        .show(q);                        -- imprime: q ==> 15
        .show(z);                        -- imprime: z ==> 16
]
